<Steps titleSize="h2">

<Step title="Get the ActivityElement">

Access the `ActivityElement` to subscribe to activities or create custom ones.

<Tabs>
<Tab title="React / Next.js">

**Using Hook:**
```jsx
import { useActivityUtils } from '@veltdev/react';

function YourComponent() {
  const activityElement = useActivityUtils();
  // activityElement is null until the Velt client is ready
}
```

**Using API:**
```jsx
import { useVeltClient } from '@veltdev/react';
import { useEffect } from 'react';

function YourComponent() {
  const { client } = useVeltClient();

  useEffect(() => {
    if (!client) return;
    const activityElement = client.getActivityElement();
  }, [client]);
}
```
</Tab>

<Tab title="Other Frameworks">
```javascript
const activityElement = Velt.getActivityElement();
```
</Tab>
</Tabs>

</Step>

<Step title="Subscribe to Activities">

Call `getAllActivities()` to start receiving real-time activity records. The observable emits the full list whenever the feed changes.

<Tabs>
<Tab title="React / Next.js">

**Using Hook:**
```jsx
import { useAllActivities } from '@veltdev/react';

function ActivityFeed() {
  const activities = useAllActivities();

  if (activities === null) return <p>Loading...</p>;
  if (activities.length === 0) return <p>No activities yet.</p>;

  return (
    <ul>
      {activities.map((activity) => (
        <li key={activity.id}>{activity.displayMessage}</li>
      ))}
    </ul>
  );
}
```

**Using API:**
```jsx
import { useVeltClient } from '@veltdev/react';
import { useEffect } from 'react';

function YourComponent() {
  const { client } = useVeltClient();

  useEffect(() => {
    if (!client) return;
    const activityElement = client.getActivityElement();
    const subscription = activityElement.getAllActivities().subscribe((activities) => {
      if (activities === null) return;
      console.log(activities);
    });
    return () => subscription?.unsubscribe();
  }, [client]);
}
```
</Tab>

<Tab title="Other Frameworks">
```javascript
const activityElement = Velt.getActivityElement();

const subscription = activityElement.getAllActivities().subscribe((activities) => {
  if (activities === null) return;
  console.log(activities.map((a) => a.displayMessage));
});

// To unsubscribe:
// subscription?.unsubscribe();
```
</Tab>
</Tabs>

</Step>
