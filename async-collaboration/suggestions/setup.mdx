## 1. Tag DOM elements as suggestion targets

Add the `data-velt-suggestion-target` attribute to any element you want to participate in suggestion mode.

<Tabs>
<Tab title="React / Next.js">
```jsx
<input
  data-velt-suggestion-target="row.123.qty"
  type="number"
  defaultValue="5"
/>
```
</Tab>

<Tab title="Other Frameworks">
```html
<input
  data-velt-suggestion-target="row.123.qty"
  type="number"
  value="5">
```
</Tab>
</Tabs>

## 2. Register suggestion targets

For complex (non-primitive) targets, register a getter so the SDK can read the current value. In React, use `useRegisterTarget()` to get a `registerTarget` callback. In other frameworks, call `registerTarget()` on the `SuggestionElement`.

<Tabs>
<Tab title="React / Next.js">
```jsx
import { useRegisterTarget, useUnregisterTarget } from '@veltdev/react';
import { useEffect } from 'react';

function EditableRow() {
  const { registerTarget } = useRegisterTarget();
  const { unregisterTarget } = useUnregisterTarget();

  useEffect(() => {
    registerTarget({
      targetId: 'row.123',
      getter: () => ({
        qty: parseInt(document.getElementById('qty-input').value, 10),
        price: parseInt(document.getElementById('price-input').value, 10),
      }),
    });
    // registerTarget returns void; remove the registration with unregisterTarget.
    return () => unregisterTarget('row.123');
  }, []);

  return (
    <div data-velt-suggestion-target="row.123">
      <input id="qty-input" type="number" defaultValue="5" />
      <input id="price-input" type="number" defaultValue="99" />
    </div>
  );
}
```
</Tab>

<Tab title="Other Frameworks">
```js
suggestionElement.registerTarget({
  targetId: 'row.123',
  getter: () => ({
    qty: parseInt(document.getElementById('qty-input').value, 10),
    price: parseInt(document.getElementById('price-input').value, 10),
  }),
});
```
</Tab>
</Tabs>

## 3. Enable suggestion mode

Activate suggestion mode with optional lifecycle callbacks. In React, use `useEnableSuggestionMode()`. In other frameworks, call `enableSuggestionMode()` on the `SuggestionElement`.

<Tabs>
<Tab title="React / Next.js">
```jsx
import { useEnableSuggestionMode, useDisableSuggestionMode } from '@veltdev/react';

function Toolbar() {
  const { enableSuggestionMode } = useEnableSuggestionMode();
  const { disableSuggestionMode } = useDisableSuggestionMode();

  const handleEnable = () => {
    enableSuggestionMode({
      onTargetEditStart: ({ targetId, oldValue, newValue, element }) => {
        // Informational. Reserved for future overrides.
      },
      onTargetEditCommit: ({ targetId, oldValue, newValue }) => {
        return {
          summary: `${targetId}: ${oldValue} → ${newValue}`,
          metadata: { source: 'inline-edit' },
        };
      },
    });
  };

  return (
    <>
      <button onClick={handleEnable}>Enable Suggestion Mode</button>
      <button onClick={() => disableSuggestionMode()}>Disable</button>
    </>
  );
}
```
</Tab>

<Tab title="Other Frameworks">
```js
suggestionElement.enableSuggestionMode({
  onTargetEditStart: ({ targetId, oldValue, newValue, element }) => {
    // Informational. Reserved for future overrides.
  },
  onTargetEditCommit: ({ targetId, oldValue, newValue }) => {
    return {
      summary: `${targetId}: ${oldValue} → ${newValue}`,
      metadata: { source: 'inline-edit' },
    };
  },
});
```
</Tab>
</Tabs>

## 4. Listen to suggestion events

Subscribe to suggestion lifecycle events. In React, use `useSuggestionEventCallback()` to reactively subscribe. In other frameworks, use `suggestionElement.on()`.

<Tabs>
<Tab title="React / Next.js">
```jsx
import { useSuggestionEventCallback } from '@veltdev/react';
import { useEffect } from 'react';

function SuggestionHandler() {
  const approved = useSuggestionEventCallback('suggestionApproved');
  const rejected = useSuggestionEventCallback('suggestionRejected');

  useEffect(() => {
    if (approved?.suggestion) {
      // Apply approved.suggestion.newValue to your app state.
    }
  }, [approved]);

  useEffect(() => {
    if (rejected?.suggestion) {
      console.log('Rejected:', rejected.suggestion);
    }
  }, [rejected]);

  return null;
}
```
</Tab>

<Tab title="Other Frameworks">
```js
suggestionElement.on('suggestionApproved').subscribe(({ suggestion }) => {
  // Apply suggestion.newValue to your app state here.
});

suggestionElement.on('suggestionRejected').subscribe(({ suggestion }) => {
  console.log('Rejected:', suggestion);
});
```
</Tab>
</Tabs>
