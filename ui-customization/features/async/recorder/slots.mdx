<Steps titleSize="h2">

<Step stepNumber="1" title="Replace the Recorder Tool">
Provide a template for the Recorder Tool.

Target the `button` slot with your own custom template.


</Step>
</Steps>


<RequestExample>

```js React / Next.js
import {
  VeltRecorderTool
} from '@veltdev/react';

export default function App() {

  return (
    <>

      <VeltRecorderTool>
        <button slot="button">Recorder</button>
      </VeltRecorderTool>

    </>
  );
}
```

```html HTML

<!doctype html>
<html lang="en">
  <head>
    <title>Recorder documentation</title>
  </head>
  <body>

    <velt-user-invite-tool>
        <button slot="button">Recorder</button>
    </velt-user-invite-tool>

  </body>
</html>
```

</RequestExample>
